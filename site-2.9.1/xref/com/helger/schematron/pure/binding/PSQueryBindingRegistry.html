19  import java.util.HashMap;
20  import java.util.Map;
21  import java.util.concurrent.locks.ReadWriteLock;
22  import java.util.concurrent.locks.ReentrantReadWriteLock;
23  
24  import javax.annotation.Nonnull;
25  import javax.annotation.Nullable;
26  import javax.annotation.concurrent.ThreadSafe;
27  
28  import com.helger.commons.ValueEnforcer;
29  import com.helger.commons.annotations.Nonempty;
30  import com.helger.commons.annotations.PresentForCodeCoverage;
31  import com.helger.commons.annotations.ReturnsMutableCopy;
32  import com.helger.commons.collections.ContainerHelper;
33  import com.helger.commons.exceptions.InitializationException;
34  import com.helger.schematron.pure.binding.xpath.PSXPathQueryBinding;

36  /**
37   * The registry class for all available query bindings. To register your own
38   * query binding, call {@link #registerQueryBinding(String, IPSQueryBinding)}.
39   *
40   * @author Philip Helger
41   */
42  @ThreadSafe
43  public final class PSQueryBindingRegistry
44  {
45    /**
46     * Name of the query binding for which the default binding is registered.
47     */
48    public static final String QUERY_BINDING_XSLT = "xslt";
49  
50    /**
51     * Name of the query binding for which the default binding is registered.
52     */
53    public static final String QUERY_BINDING_XSLT2 = "xslt2";
54  
55    /**
56     * The default XPath binding object to be used
57     */
58    public static final IPSQueryBinding DEFAULT_QUERY_BINDING = PSXPathQueryBinding.getInstance ();
59  
60    private static final ReadWriteLock s_aRWLock = new ReentrantReadWriteLock ();
61    private static final Map <String, IPSQueryBinding> s_aMap = new HashMap <String, IPSQueryBinding> ();
62  
63    static
64    {
65      try
66      {
67        registerQueryBinding (QUERY_BINDING_XSLT, DEFAULT_QUERY_BINDING);
68        registerQueryBinding (QUERY_BINDING_XSLT2, DEFAULT_QUERY_BINDING);
69      }
70      catch (final SchematronBindException ex)
71      {
72        throw new InitializationException (ex);
73      }
74    }
75  
76    @PresentForCodeCoverage
77    @SuppressWarnings ("unused")
78    private static final PSQueryBindingRegistry s_aInstance = new PSQueryBindingRegistry ();
79  
80    private PSQueryBindingRegistry ()
81    {}
82  
83    public static void registerQueryBinding (@Nonnull @Nonempty final String sName,
84                                             @Nonnull final IPSQueryBinding aQueryBinding) throws SchematronBindException
85    {
86      ValueEnforcer.notEmpty (sName, "Name");
87      ValueEnforcer.notNull (aQueryBinding, "QueryBinding");
88  
89      s_aRWLock.writeLock ().lock ();
90      try
91      {
92        if (s_aMap.containsKey (sName))
93          throw new SchematronBindException ("A queryBinding with the name '" + sName + "' is already registered!");
94        s_aMap.put (sName, aQueryBinding);
95      }
96      finally
97      {
98        s_aRWLock.writeLock ().unlock ();
99      }
100   }
101 
102   /**
103    * Get the query binding with the specified name.
104    *
105    * @param sName
106    *        The name to the query binding to retrieve. May be <code>null</code>.
107    *        If it is <code>null</code> than the {@link #DEFAULT_QUERY_BINDING}
108    *        object is returned.
109    * @return <code>null</code> if no such query binding was found.
110    */
111   @Nullable
112   public static IPSQueryBinding getQueryBindingOfName (@Nullable final String sName)
113   {
114     if (sName == null)
115       return DEFAULT_QUERY_BINDING;
116 
117     s_aRWLock.readLock ().lock ();
118     try
119     {
120       return s_aMap.get (sName);
121     }
122     finally
123     {
124       s_aRWLock.readLock ().unlock ();
125     }
126   }
127 
128   /**
129    * Get the query binding with the specified name
130    *
131    * @param sName
132    *        The name of the query binding to look up. May be <code>null</code>.
133    * @return Never <code>null</code>.
134    * @throws SchematronBindException
135    *         In case the query binding could not be resolved!
136    */
137   @Nonnull
138   public static IPSQueryBinding getQueryBindingOfNameOrThrow (@Nullable final String sName) throws SchematronBindException
139   {
140     final IPSQueryBinding aQB = getQueryBindingOfName (sName);
141     if (aQB == null)
142       throw new SchematronBindException ("No query binding implementation present for query binding '" + sName + "'");
143     return aQB;
144   }
145 
146   /**
147    * @return A non-<code>null</code> map with all contained query bindings from
148    *         name to object.
149    */
150   @Nonnull
151   @ReturnsMutableCopy
152   public static Map <String, IPSQueryBinding> getAllRegisteredQueryBindings ()
153   {
154     s_aRWLock.readLock ().lock ();
155     try
156     {
157       return ContainerHelper.newMap (s_aMap);
158     }
159     finally
160     {
161       s_aRWLock.readLock ().unlock ();
162     }
163   }
164 }
