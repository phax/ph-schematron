17  package com.helger.schematron.svrl;
18  
19  import javax.annotation.Nonnull;
20  import javax.annotation.Nullable;
21  
22  import com.helger.commons.annotations.IsSPIInterface;
23  
24  /**
25   * Implement this SPI interface to beautify SVRL error messages by replacing
26   * namespaceURLs with common prefixes.
27   *
28   * @author Philip Helger
29   */
30  @IsSPIInterface
31  public interface ISVRLLocationBeautifierSPI
32  {
33    /**
34     * Beautify the passed combination.
35     *
36     * @param sNamespaceURI
37     *        Namespace URI
38     * @param sLocalName
39     *        Element local name
40     * @return <code>null</code> to indicate that this object does not know how to
41     *         handle the namespace. Otherwise the replacement text should be
42     *         returned.<br>
43     *         Example for UBL: Input parameters
44     *         <code>urn:oasis:names:specification:ubl:schema:xsd:CommonAggregateComponents-2</code>
45     *         and <code>Party</code>. The result may be <code>cac:Party</code> as
46     *         <i>cac</i> is the common namespace prefix for the passed namespace
47     *         URI.
48     */
49    @Nullable
50    String getReplacementText (@Nonnull String sNamespaceURI, @Nonnull String sLocalName);
51  }
