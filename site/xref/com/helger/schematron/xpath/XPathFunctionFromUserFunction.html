17  package com.helger.schematron.xpath;
18  
19  import java.util.List;
20  
21  import javax.annotation.Nonnull;
22  import javax.annotation.Nullable;
23  import javax.xml.xpath.XPathFunction;
24  import javax.xml.xpath.XPathFunctionException;
25  
26  import net.sf.saxon.Configuration;
27  import net.sf.saxon.Controller;
28  import net.sf.saxon.expr.JPConverter;
29  import net.sf.saxon.expr.XPathContextMajor;
30  import net.sf.saxon.expr.instruct.UserFunction;
31  import net.sf.saxon.om.Sequence;
32  
33  import com.helger.commons.ValueEnforcer;
34  
35  /**
36   * A proxy for an {@link XPathFunction} that is implemented as Saxon
37   * {@link UserFunction}.
38   *
39   * @author Philip Helger
40   */
41  public final class XPathFunctionFromUserFunction implements XPathFunction
42  {
43    private final Configuration m_aConfiguration;
44    private final Controller m_aXQController;
45    private final UserFunction m_aUserFunc;
46    private final XPathContextMajor m_aXPathContext;
47  
48    public XPathFunctionFromUserFunction (@Nonnull final Configuration aConfiguration,
49                                          @Nonnull final Controller aXQController,
50                                          @Nonnull final UserFunction aUserFunc)
51    {
52      m_aConfiguration = ValueEnforcer.notNull (aConfiguration, "Configuration");
53      m_aUserFunc = ValueEnforcer.notNull (aUserFunc, "UserFunc");
54      m_aXQController = ValueEnforcer.notNull (aXQController, "XQController");
55      // This is surely not correct, but it works :)
56      m_aXPathContext = aXQController.newXPathContext ();
57    }
58  
59    @Nullable
60    public Object evaluate (@SuppressWarnings ("rawtypes") final List args) throws XPathFunctionException
61    {
62      try
63      {
64        // Convert the parameters
65        final Sequence [] aValues = new Sequence [args.size ()];
66        int i = 0;
67        for (final Object arg : args)
68        {
69          // Ripped from Saxon itself
70          final JPConverter converter = JPConverter.allocate (arg.getClass (), m_aConfiguration);
71          aValues[i] = converter.convert (arg, m_aXPathContext);
72          ++i;
73        }
74        // Invoke function
75        return m_aUserFunc.call (aValues, m_aXQController);
76      }
77      catch (final Exception ex)
78      {
79        // Wrap all exceptions
80        throw new XPathFunctionException (ex);
81      }
82    }
83  }
