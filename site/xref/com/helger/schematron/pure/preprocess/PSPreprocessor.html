49  /**
50   * This is the pre-processor class for pure Schematron. It converts an existing
51   * schema to the minimal syntax (by default) but allows for a certain degree of
52   * customization by keeping certain elements in the resulting schema. The actual
53   * query binding is used, so that report test expressions can be converted to
54   * assertions, and to replace the content of &lt;param&gt; elements into actual
55   * values.
56   *
57   * @author Philip Helger
58   */
59  @NotThreadSafe
60  public class PSPreprocessor
61  {

64    public static final boolean DEFAULT_KEEP_REPORTS = false;
65    public static final boolean DEFAULT_KEEP_EMPTY_PATTERNS = true;
66    public static final boolean DEFAULT_KEEP_EMPTY_SCHEMA = true;

178   @Nonnull
179   private static PSPhase _getPreprocessedPhase (@Nonnull final PSPhase aPhase, @Nonnull final PreprocessorIDPool aIDPool) throws SchematronPreprocessException
180   {
181     final PSPhase ret = new PSPhase ();
182     ret.setID (aIDPool.getUniqueID (aPhase.getID ()));
183     ret.setRich (aPhase.getRichClone ());
184     if (aPhase.hasAnyInclude ())
185       throw new SchematronPreprocessException ("Cannot preprocess <phase> with an <include>");
186     for (final IPSElement aElement : aPhase.getAllContentElements ())
187     {
188       if (aElement instanceof PSActive)
189         ret.addActive (((PSActive) aElement).getClone ());
190       else
191         if (aElement instanceof PSLet)
192           ret.addLet (((PSLet) aElement).getClone ());
193       // ps are ignored
194     }
195     ret.addForeignElements (aPhase.getAllForeignElements ());
196     ret.addForeignAttributes (aPhase.getAllForeignAttributes ());
197     return ret;
198   }

200   /**
201    * Resolve all &lt;extends&gt; elements. This method calls itself recursively
202    * until all extends elements are resolved.
203    *
204    * @param aRuleContent
205    *        A list consisting of {@link PSAssertReport} and {@link PSExtends}
206    *        objects. Never <code>null</code>.
207    * @param aLookup
208    *        The rule lookup object
209    * @return List of assert/report elements. Never <code>null</code>.
210    * @throws SchematronPreprocessException
211    *         If the base rule of an extends object could not be resolved.
212    */
213   @Nonnull
214   @ReturnsMutableCopy
215   private static List <PSAssertReport> _getResolvedExtends (@Nonnull final List <IPSElement> aRuleContent,
216                                                             @Nonnull final PreprocessorLookup aLookup) throws SchematronPreprocessException
217   {
218     final List <PSAssertReport> ret = new ArrayList <PSAssertReport> ();
219     for (final IPSElement aElement : aRuleContent)
220     {
221       if (aElement instanceof PSAssertReport)
222         ret.add ((PSAssertReport) aElement);
223       else
224       {
225         final PSExtends aExtends = (PSExtends) aElement;
226         final String sRuleID = aExtends.getRule ();
227         final PSRule aBaseRule = aLookup.getAbstractRuleOfID (sRuleID);
228         if (aBaseRule == null)
229           throw new SchematronPreprocessException ("Failed to resolve rule ID '" +
230                                                    sRuleID +
231                                                    "' in extends statement. Available rules are: " +
232                                                    aLookup.getAllAbstractRuleIDs ());
233         // Recursively resolve the extends of the base rule
234         ret.addAll (_getResolvedExtends (aBaseRule.getAllContentElements (), aLookup));
235       }
236     }
237     return ret;
238   }

240   @Nonnull
241   private PSAssertReport _getPreprocessedAssert (@Nonnull final PSAssertReport aAssertReport,
242                                                  @Nonnull final PreprocessorIDPool aIDPool,
243                                                  @Nullable final Map <String, String> aParamValueMap)
244   {
245     String sTest = aAssertReport.getTest ();
246     if (aAssertReport.isReport () && !m_bKeepReports)
247     {
248       // Negate the expression!
249       sTest = m_aQueryBinding.getNegatedTestExpression (sTest);
250     }
251 
252     // Keep report or make it always an assert
253     final PSAssertReport ret = new PSAssertReport (m_bKeepReports ? aAssertReport.isAssert () : true);
254     ret.setTest (m_aQueryBinding.getWithParamTextsReplaced (sTest, aParamValueMap));
255     ret.setFlag (aAssertReport.getFlag ());
256     ret.setID (aIDPool.getUniqueID (aAssertReport.getID ()));
257     if (m_bKeepDiagnostics)
258       ret.setDiagnostics (aAssertReport.getAllDiagnostics ());
259     ret.setRich (aAssertReport.getRichClone ());
260     ret.setLinkable (aAssertReport.getLinkableClone ());
261     for (final Object aContent : aAssertReport.getAllContentElements ())
262     {
263       if (aContent instanceof String)
264         ret.addText ((String) aContent);
265       else
266         if (aContent instanceof PSName)
267           ret.addName (((PSName) aContent).getClone ());
268         else
269           if (aContent instanceof PSValueOf)
270           {
271             final PSValueOf aValueOf = ((PSValueOf) aContent).getClone ();
272             aValueOf.setSelect (m_aQueryBinding.getWithParamTextsReplaced (aValueOf.getSelect (), aParamValueMap));
273             ret.addValueOf (aValueOf);
274           }
275           else
276             if (aContent instanceof PSEmph)
277               ret.addEmph (((PSEmph) aContent).getClone ());
278             else
279               if (aContent instanceof PSDir)
280                 ret.addDir (((PSDir) aContent).getClone ());
281               else
282                 if (aContent instanceof PSSpan)
283                   ret.addSpan (((PSSpan) aContent).getClone ());
284     }
285     ret.addForeignElements (aAssertReport.getAllForeignElements ());
286     ret.addForeignAttributes (aAssertReport.getAllForeignAttributes ());
287     return ret;
288   }

290   @Nullable
291   private PSRule _getPreprocessedRule (@Nonnull final PSRule aRule,
292                                        @Nonnull final PreprocessorLookup aLookup,
293                                        @Nonnull final PreprocessorIDPool aIDPool,
294                                        @Nullable final Map <String, String> aParamValueMap) throws SchematronPreprocessException
295   {
296     if (aRule.isAbstract ())
297     {
298       // Will be inlined
299       return null;
300     }
301 
302     final PSRule ret = new PSRule ();
303     ret.setFlag (aRule.getFlag ());
304     ret.setRich (aRule.getRichClone ());
305     ret.setLinkable (aRule.getLinkableClone ());
306     // abstract is always false
307     ret.setContext (m_aQueryBinding.getWithParamTextsReplaced (aRule.getContext (), aParamValueMap));
308     ret.setID (aIDPool.getUniqueID (aRule.getID ()));
309     if (aRule.hasAnyInclude ())
310       throw new SchematronPreprocessException ("Cannot preprocess <rule> with an <include>");
311     for (final PSLet aLet : aRule.getAllLets ())
312       ret.addLet (aLet.getClone ());
313     for (final PSAssertReport aAssertReport : _getResolvedExtends (aRule.getAllContentElements (), aLookup))
314       ret.addAssertReport (_getPreprocessedAssert (aAssertReport, aIDPool, aParamValueMap));
315     ret.addForeignElements (aRule.getAllForeignElements ());
316     ret.addForeignAttributes (aRule.getAllForeignAttributes ());
317     return ret;
318   }

320   @Nullable
321   private PSPattern _getPreprocessedPattern (@Nonnull final PSPattern aPattern,
322                                              @Nonnull final PreprocessorLookup aLookup,
323                                              @Nonnull final PreprocessorIDPool aIDPool) throws SchematronPreprocessException
324   {
325     if (aPattern.isAbstract ())
326     {
327       // Will be inlined
328       return null;
329     }
330 
331     final PSPattern ret = new PSPattern ();
332     // abstract always false
333     // is-a must be resolved
334     ret.setID (aIDPool.getUniqueID (aPattern.getID ()));
335     ret.setRich (aPattern.getRichClone ());
336     if (aPattern.hasAnyInclude ())
337       throw new SchematronPreprocessException ("Cannot preprocess <pattern> with an <include>");
338     if (m_bKeepTitles && aPattern.hasTitle ())
339       ret.setTitle (aPattern.getTitle ().getClone ());
340 
341     final String sIsA = aPattern.getIsA ();
342     if (sIsA != null)
343     {
344       final PSPattern aBasePattern = aLookup.getAbstractPatternOfID (sIsA);
345       if (aBasePattern == null)
346         throw new SchematronPreprocessException ("Failed to resolve the pattern denoted by is-a='" + sIsA + "'");
347 
348       if (!ret.hasID ())
349         ret.setID (aIDPool.getUniqueID (aBasePattern.getID ()));
350       if (!ret.hasRich ())
351         ret.setRich (aBasePattern.getRichClone ());
352 
353       // get the string replacements
354       final Map <String, String> aParamValueMap = m_aQueryBinding.getStringReplacementMap (aPattern.getAllParams ());
355 
356       for (final IPSElement aElement : aBasePattern.getAllContentElements ())
357       {
358         if (aElement instanceof PSLet)
359           ret.addLet (((PSLet) aElement).getClone ());
360         else
361           if (aElement instanceof PSRule)
362           {
363             final PSRule aMinifiedRule = _getPreprocessedRule ((PSRule) aElement, aLookup, aIDPool, aParamValueMap);
364             if (aMinifiedRule != null)
365               ret.addRule (aMinifiedRule);
366           }
367         // params must have be resolved
368         // ps are ignored
369       }
370     }
371     else
372     {
373       for (final IPSElement aElement : aPattern.getAllContentElements ())
374       {
375         if (aElement instanceof PSLet)
376           ret.addLet (((PSLet) aElement).getClone ());
377         else
378           if (aElement instanceof PSRule)
379           {
380             final PSRule aMinifiedRule = _getPreprocessedRule ((PSRule) aElement, aLookup, aIDPool, null);
381             if (aMinifiedRule != null)
382               ret.addRule (aMinifiedRule);
383           }
384         // params must be resolved
385         // ps are ignored
386       }
387     }
388     ret.addForeignElements (aPattern.getAllForeignElements ());
389     ret.addForeignAttributes (aPattern.getAllForeignAttributes ());
390     return ret;
391   }

406   /**
407    * Convert the passed schema to a minimal schema.
408    *
409    * @param aSchema
410    *        The schema to be made minimal. May not be <code>null</code>
411    * @return The original schema object, if it is already minimal - a minimal
412    *         copy otherwise! May be <code>null</code> if the original schema is
413    *         not yet minimal and {@link #isKeepEmptySchema()} is set to
414    *         <code>false</code>.
415    * @throws SchematronPreprocessException
416    *         In case a preprocessing error occurs
417    */
418   @Nullable
419   public PSSchema getAsMinimalSchema (@Nonnull final PSSchema aSchema) throws SchematronPreprocessException
420   {
421     ValueEnforcer.notNull (aSchema, "Schema");
422 
423     // Anything to do?
424     if (aSchema.isMinimal ())
425       return aSchema;
426 
427     return getForcedPreprocessedSchema (aSchema);
428   }

454   /**
455    * Convert the passed schema to a pre-processed schema independent if it is
456    * already minimal or not.
457    *
458    * @param aSchema
459    *        The schema to be made minimal. May not be <code>null</code>
460    * @return A minimal copy of the schema. May be <code>null</code> if the
461    *         original schema is not yet minimal and {@link #isKeepEmptySchema()}
462    *         is set to <code>false</code>.
463    * @throws SchematronPreprocessException
464    *         In case a preprocessing error occurs
465    */
466   @Nullable
467   public PSSchema getForcedPreprocessedSchema (@Nonnull final PSSchema aSchema) throws SchematronPreprocessException
468   {
469     ValueEnforcer.notNull (aSchema, "Schema");
470 
471     final PreprocessorLookup aLookup = new PreprocessorLookup (aSchema);
472     final PreprocessorIDPool aIDPool = new PreprocessorIDPool ();
473 
474     final PSSchema ret = new PSSchema (aSchema.getResource ());
475     ret.setID (aIDPool.getUniqueID (aSchema.getID ()));
476     ret.setRich (aSchema.getRichClone ());
477     ret.setSchemaVersion (aSchema.getSchemaVersion ());
478     ret.setDefaultPhase (aSchema.getDefaultPhase ());
479     ret.setQueryBinding (aSchema.getQueryBinding ());
480     if (m_bKeepTitles && aSchema.hasTitle ())
481       ret.setTitle (aSchema.getTitle ().getClone ());
482     if (aSchema.hasAnyInclude ())
483       throw new SchematronPreprocessException ("Cannot preprocess <schema> with an <include>");
484     for (final PSNS aNS : aSchema.getAllNSs ())
485       ret.addNS (aNS.getClone ());
486     // start ps are skipped
487     for (final PSLet aLet : aSchema.getAllLets ())
488       ret.addLet (aLet.getClone ());
489     for (final PSPhase aPhase : aSchema.getAllPhases ())
490       ret.addPhase (_getPreprocessedPhase (aPhase, aIDPool));
491     for (final PSPattern aPattern : aSchema.getAllPatterns ())
492     {
493       final PSPattern aMinifiedPattern = _getPreprocessedPattern (aPattern, aLookup, aIDPool);
494       if (aMinifiedPattern != null)
495       {
496         // Pattern without rules?
497         if (aMinifiedPattern.getRuleCount () > 0 || m_bKeepEmptyPatterns)
498           ret.addPattern (aMinifiedPattern);
499       }
500     }
501 
502     // Schema without patterns?
503     if (aSchema.getPatternCount () == 0 && !m_bKeepEmptySchema)
504       return null;
505 
506     // end ps are skipped
507     if (m_bKeepDiagnostics && aSchema.hasDiagnostics ())
508       ret.setDiagnostics (_getPreprocessedDiagnostics (aSchema.getDiagnostics ()));
509     ret.addForeignElements (aSchema.getAllForeignElements ());
510     ret.addForeignAttributes (aSchema.getAllForeignAttributes ());
511     return ret;
512   }

514   @Override
515   public String toString ()
516   {
517     return new ToStringGenerator (this).append ("queryBinding", m_aQueryBinding)
518                                        .append ("keepTitles", m_bKeepTitles)
519                                        .append ("keepDiagnostics", m_bKeepDiagnostics)
520                                        .append ("keepReports", m_bKeepReports)
521                                        .append ("keepEmptyPatterns", m_bKeepEmptyPatterns)
522                                        .append ("keepEmptySchema", m_bKeepEmptySchema)
523                                        .toString ();
524   }
