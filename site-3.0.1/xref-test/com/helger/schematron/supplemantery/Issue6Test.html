44    public static void validateAndProduceSVRL (final File schematron, final File xml) throws Exception
45    {
46      final IReadableResource aSchematron = new FileSystemResource (schematron.getAbsoluteFile ());
47      final IReadableResource anXMLSource = new FileSystemResource (xml.getAbsoluteFile ());
48      final AbstractSchematronResource aSCH = new SchematronResourceSCH (aSchematron);
49      if (aSCH instanceof SchematronResourcePure)
50        ((SchematronResourcePure) aSCH).setErrorHandler (new LoggingPSErrorHandler ());
51      else
52        System.out.println (XMLWriter.getXMLString (((SchematronResourceSCH) aSCH).getXSLTProvider ().getXSLTDocument ()));
53      final SchematronOutputType aSVRL = aSCH.applySchematronValidationToSVRL (anXMLSource);
54      assertNotNull (aSVRL);
55      System.out.println (SVRLWriter.createXMLString (aSVRL));
56    }
