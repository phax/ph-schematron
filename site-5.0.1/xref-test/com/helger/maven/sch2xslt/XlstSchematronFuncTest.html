17  package com.helger.maven.sch2xslt;
18  
19  import java.io.File;
20  import java.io.IOException;
21  
22  import org.junit.Ignore;
23  import org.junit.Test;
24  
25  import com.helger.commons.io.resource.ClassPathResource;
26  import com.helger.schematron.xslt.SchematronProviderXSLTFromSCH;
27  
28  import net.sf.saxon.Transform;
29  
30  public final class XlstSchematronFuncTest
31  {
32    private static final String TEST_XSLT_SCHEMATRON = "testXsltSchematron";
33  
34    @Test
35    @Ignore ("Does not work on the commandline, because the XSLT resources are only available in the classpath")
36    public void testXsltSchematron () throws IOException
37    {
38      final Transform saxonTransform = new Transform ();
39  
40      final String sourcePath = new File ("src/test/resources/schematron/check-classifications.sch").getCanonicalPath ();
41      final String step1Path = new ClassPathResource (SchematronProviderXSLTFromSCH.XSLT2_STEP1).getAsURL ()
42                                                                                                .toExternalForm ();
43      final String step2Path = new ClassPathResource (SchematronProviderXSLTFromSCH.XSLT2_STEP2).getAsURL ()
44                                                                                                .toExternalForm ();
45      final String step3Path = new ClassPathResource (SchematronProviderXSLTFromSCH.XSLT2_STEP3).getAsURL ()
46                                                                                                .toExternalForm ();
47      final String outputPath = new File ("target/test/schematron-via-xslt/").getCanonicalPath ();
48  
49      final String [] step1Commands = { "-xsl:" + step1Path, "-s:" + sourcePath, "-o:" + outputPath + "/step1.sch" };
50      final String [] step2Commands = { "-xsl:" +
51                                        step2Path,
52                                        "-s:" + outputPath + "/step1.sch",
53                                        "-o:" + outputPath + "/step2.sch" };
54      final String [] step3Commands = { "-xsl:" +
55                                        step3Path,
56                                        "-s:" + outputPath + "/step2.sch",
57                                        "-o:" + outputPath + "/check-classifications.xslt" };
58  
59      saxonTransform.doTransform (step1Commands, TEST_XSLT_SCHEMATRON);
60      saxonTransform.doTransform (step2Commands, TEST_XSLT_SCHEMATRON);
61      saxonTransform.doTransform (step3Commands, TEST_XSLT_SCHEMATRON);
62    }
63  }
