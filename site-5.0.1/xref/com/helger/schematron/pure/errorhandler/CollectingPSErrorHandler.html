17  package com.helger.schematron.pure.errorhandler;
18  
19  import javax.annotation.Nullable;
20  
21  import com.helger.commons.error.list.ErrorList;
22  
23  /**
24   * An implementation if {@link IPSErrorHandler} that collects all error
25   * messages.
26   *
27   * @author Philip Helger
28   */
29  public class CollectingPSErrorHandler extends AbstractCollectingPSErrorHandler
30  {
31    public CollectingPSErrorHandler ()
32    {
33      this (null);
34    }
35  
36    public CollectingPSErrorHandler (@Nullable final IPSErrorHandler aNestedErrorHandler)
37    {
38      super (new ErrorList (), aNestedErrorHandler);
39    }
40  }
