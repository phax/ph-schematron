42  /**
43   * Helper class that validates a Schematron against the RelaxNG Compact scheme.
44   * This class is not suitable for validating an XML instance against a
45   * Schematron instance.
46   *
47   * @author Philip Helger
48   */
49  @Immutable
50  public final class SchematronValidator
51  {
52    /** Path to the Schematron RelaxNG Compact file within the class path */
53    public static final String SCHEMATRON_RNC_PATH = "schemas/iso-schematron-2006.rnc";
54  
55    /** The readable resource with the RelaxNG compact Schematron rules */
56    public static final ClassPathResource SCHEMATRON_RNC = new ClassPathResource (SCHEMATRON_RNC_PATH);
57  
58    private static final Logger s_aLogger = LoggerFactory.getLogger (SchematronValidator.class);
59  
60    @PresentForCodeCoverage
61    private static final SchematronValidator s_aInstance = new SchematronValidator ();
62  
63    private SchematronValidator ()
64    {}
65  
66    /**
67     * Check if the passed micro node is a valid schematron instance.
68     *
69     * @param aNode
70     *        The micro node to check. May be <code>null</code>.
71     * @return <code>true</code> if the schematron is valid, <code>false</code>
72     *         otherwise.
73     */
74    public static boolean isValidSchematron (@Nullable final IMicroNode aNode)
75    {
76      if (aNode == null)
77        return false;
78  
79      return isValidSchematron (TransformSourceFactory.create (MicroWriter.getNodeAsString (aNode)));
80    }
81  
82    /**
83     * Check if the passed DOM node is a valid schematron instance.
84     *
85     * @param aNode
86     *        The DOM node to check. May be <code>null</code>.
87     * @return <code>true</code> if the schematron is valid, <code>false</code>
88     *         otherwise.
89     */
90    public static boolean isValidSchematron (@Nullable final Node aNode)
91    {
92      if (aNode == null)
93        return false;
94  
95      return isValidSchematron (TransformSourceFactory.create (aNode));
96    }
97  
98    /**
99     * Check if the passed resource is a valid schematron instance.
100    *
101    * @param aRes
102    *        The resource to check. May be <code>null</code>.
103    * @return <code>true</code> if the schematron is valid, <code>false</code>
104    *         otherwise.
105    */
106   public static boolean isValidSchematron (@Nullable final IReadableResource aRes)
107   {
108     if (aRes == null)
109       return false;
110 
111     return isValidSchematron (TransformSourceFactory.create (aRes));
112   }
113 
114   /**
115    * Check if the passed source is a valid schematron instance.
116    *
117    * @param aSource
118    *        The source to check. May be <code>null</code>.
119    * @return <code>true</code> if the schematron is valid, <code>false</code>
120    *         otherwise.
121    */
122   public static boolean isValidSchematron (@Nullable final Source aSource)
123   {
124     if (aSource == null)
125       return false;
126 
127     try
128     {
129       // Get a validator from the RNC schema.
130       final Validator aValidator = RelaxNGCompactSchemaCache.getInstance ().getValidator (SCHEMATRON_RNC);
131 
132       // Ensure a collecting error handler is set
133       final ErrorHandler aOldEH = aValidator.getErrorHandler ();
134       CollectingSAXErrorHandler aCEH;
135       if (aOldEH instanceof CollectingSAXErrorHandler)
136         aCEH = (CollectingSAXErrorHandler) aOldEH;
137       else
138       {
139         aCEH = new CollectingSAXErrorHandler ();
140         aValidator.setErrorHandler (aCEH.andThen (aOldEH));
141       }
142 
143       // Perform validation
144       aValidator.validate (aSource, null);
145 
146       // Check results
147       return aCEH.getErrorList ().getMostSevereErrorLevel ().isLT (EErrorLevel.ERROR);
148     }
149     catch (final SAXException ex)
150     {
151       // Document is not valid in respect to XML
152     }
153     catch (final IOException ex)
154     {
155       s_aLogger.warn ("Failed to read source " + aSource, ex);
156     }
157     return false;
158   }
159 }
